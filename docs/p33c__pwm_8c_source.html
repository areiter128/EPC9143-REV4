251 volatile uint16_t p33c_PwmGenerator_Initialize(volatile uint16_t pgInstance)
252 {
253  volatile uint16_t retval=1;
254  volatile struct P33C_PWM_GENERATOR_s* pg;
255 
256 
257  // Set pointer to memory address of desired PWM instance
258  pg = p33c_PwmGenerator_GetHandle(pgInstance);
259 
260  // Disable the PWM generator
261  retval &= p33c_PwmGenerator_Disable(pg);
262 
263  // Reset all SFRs to default
264  p33c_PwmGenerator_ConfigWrite(pgInstance, pgConfigClear);
265 
266  /* PWM GENERATOR CONTROL REGISTER LOW */
267  pg->PGxCONL.bits.ON = 0; // Disable PWM generator
268  pg->PGxCONL.bits.CLKSEL = 0b01; // Clock Selection: Selected by PWM module register PCLKCON.MCLKSEL bits
269  pg->PGxCONL.bits.MODSEL = 0b000; // Mode Selection: Independent Edge PWM mode
270  pg->PGxCONL.bits.TRGCNT = 0b000; // Trigger Count Selection: PWM Generator produces 1 PWM cycle after triggered
271  pg->PGxCONL.bits.HREN = 1; // PWM Generator 1 High-Resolution Enable bit: PWM Generator 1 operates in High-Resolution mode
272 
273  return(retval);
274 }

315 volatile uint16_t p33c_PwmGenerator_Enable(volatile struct P33C_PWM_GENERATOR_s* pg)
316 {
317  volatile uint16_t retval=1;
318  volatile uint16_t timeout=0;
319 
320  // Set PWM generator override bits to prevent signals being generated outside the device
321  pg->PGxIOCONL.bits.OVRENH = 1;
322  pg->PGxIOCONL.bits.OVRENL = 1;
323 
324  // Assign GPIO ownership to I/O module control
325  pg->PGxIOCONH.bits.PENH = 0;
326  pg->PGxIOCONH.bits.PENL = 0;
327 
328  // Turn on the PWM generator
329  pg->PGxCONL.bits.ON = 1;
330 
331  // enforce update of timing registers
332  pg->PGxSTAT.bits.UPDREQ = 1;
333 
334  // If high resolution mode is enabled, check if clock has locked in without errors
335  if(pg->PGxCONL.bits.HREN)
336  {
337  while((!PCLKCONbits.HRRDY) && (timeout++<5000));
338  if ((timeout >= 5000) || (PCLKCONbits.HRERR)) // if there is an error
339  return(0); // return ERROR
340 
341  }
342 
343  // Assign GPIO ownership to given PWM generator
344  pg->PGxIOCONH.bits.PENH = 1;
345  pg->PGxIOCONH.bits.PENL = 1;
346 
347  return(retval);
348 
349 }

571 volatile uint16_t p33c_PwmGenerator_GetInstance(volatile struct P33C_PWM_GENERATOR_s* pg)
572 {
573  volatile uint16_t retval=1;
574 
575  // Null-pointer protection
576  if (pg == NULL)
577  return(0);
578 
579  // Capture Instance: set pointer to memory address of desired PWM instance
580  retval = (volatile uint16_t)
581  (((volatile uint16_t)&pg->PGxCONL - (volatile uint16_t)&PG1CONL) / P33C_PWMGEN_SFR_OFFSET) + 1;
582 
583  if (retval > P33C_PG_COUNT)
584  return(0); // PWM generator not member of a valid group
585 
586  return(retval);
587 }

600 volatile uint16_t p33c_PwmGenerator_GetGroup(volatile struct P33C_PWM_GENERATOR_s* pg)
601 {
602  volatile uint16_t retval=1;
603  volatile uint16_t pgInstance;
604 
605  // Null-pointer protection
606  if (pg == NULL)
607  return(0);
608 
609  // Get group of PWM generator
610  pgInstance = (volatile uint16_t)
611  (((volatile uint16_t)&pg->PGxCONL - (volatile uint16_t)&PG1CONL) / P33C_PWMGEN_SFR_OFFSET + 1);
612 
613  // Verify PWM generator group is valid and available
614  if (pgInstance > P33C_PG_COUNT)
615  return(0); // PWM generator not member of a valid group
616  else if (pgInstance > 4)
617  retval = 2; // PWM generator is member of group #2 [PG5-PG8]
618  else
619  retval = 1; // PWM generator is member of group #1 [PG1-PG4]
620 
621  return(retval);
622 }

657 volatile uint16_t p33c_PwmGenerator_SyncGenerators(
658  volatile struct P33C_PWM_GENERATOR_s* pgHandleMother,
659  volatile uint16_t pgMotherTriggerOutput,
660  volatile struct P33C_PWM_GENERATOR_s* pgHandleChild,
661  volatile bool ChildImmediateUpdate
662  )
663 {
664  volatile uint16_t retval=1;
665  volatile uint16_t pgMotherInstance=0, pgChildInstance=0;
666  volatile uint16_t pgMotherGroup=0, pgChildGroup=0;
667  volatile uint16_t pgInstance;
668 
669  // Null-pointer protection
670  if ((pgHandleMother == NULL) || (pgHandleChild == NULL))
671  return(0);
672 
673  // Capture PWM generator instances and groups for given handles
674  pgMotherInstance = p33c_PwmGenerator_GetInstance(pgHandleMother);
675  pgMotherGroup = p33c_PwmGenerator_GetGroup(pgHandleMother);
676 
677  pgChildInstance = p33c_PwmGenerator_GetInstance(pgHandleChild);
678  pgChildGroup = p33c_PwmGenerator_GetGroup(pgHandleChild);
679 
680  // Enable update trigger broadcast in Mother PWM
681  // PWM generator broadcasts software set/clear of the UPDREQ status bit and EOC signal
682  pgHandleMother->PGxCONH.bits.MSTEN = 1;
683 
684  // PWM Generator Trigger Output Selection
685  // 0b011 = PGxTRIGC compare event is the PWM Generator trigger
686  // 0b010 = PGxTRIGB compare event is the PWM Generator trigger
687  // 0b001 = PGxTRIGA compare event is the PWM Generator trigger
688  // 0b000 = EOC event is the PWM Generator trigger
689  pgHandleMother->PGxEVTL.bits.PGTRGSEL = (pgMotherTriggerOutput & 0x0003);
690 
691  // Configure child PWM in slaved mode, incorporating immediate of EOC selection
692  pgHandleChild->PGxCONH.bits.UPDMOD = (0b010 | ChildImmediateUpdate); // Slaved SOC update
693  pgHandleChild->PGxCONH.bits.TRGMOD = 1; // PWM Generator operates in Retriggerable mode
694  pgHandleChild->PGxEVTL.bits.UPDTRG = 0b00; // Timing register updates triggered through UPDREQ bit (PGxSTAT[3])
695  pgHandleChild->PGxCONL.bits.MODSEL = 0b000; // Independent Edge PWM mode;
696 
697  /* Set sync trigger input to Child generator
698 
699  SOCS[3:0]: Start-of-Cycle Selection bits(1,2,3)
700  1111 = TRIG bit or PCI Sync function only (no hardware trigger source is selected)
701  1110-0101 = Reserved
702  0100 = Trigger output selected by PG4 or PG8 PGTRGSEL[2:0] bits (PGxEVTL[2:0])
703  0011 = Trigger output selected by PG3 or PG7 PGTRGSEL[2:0] bits (PGxEVTL[2:0])
704  0010 = Trigger output selected by PG2 or PG6 PGTRGSEL[2:0] bits (PGxEVTL[2:0])
705  0001 = Trigger output selected by PG1 or PG5 PGTRGSEL[2:0] bits (PGxEVTL[2:0])
706  0000 = Local EOC ? PWM Generator is self-triggered
707 
708  */
709 
710  if (pgMotherGroup == pgChildGroup)
711  {
712  // If both PWM generators are member of the same group,
713  // direct synchronization can be used
714 
715  if (pgMotherGroup == 1)
716  pgHandleChild->PGxCONH.bits.SOCS = pgMotherInstance;
717  else if (pgMotherGroup == 2)
718  pgHandleChild->PGxCONH.bits.SOCS = (pgMotherInstance - 4);
719  else
720  return(0); // Exit if PWM generator group is out of range
721 
722  }
723  else
724  {
725  // Synchronization across PWM generator groups need to be routed
726  // through the PCI Sync function
727 
728  pgInstance = p33c_PwmGenerator_GetInstance(pgHandleChild);
729 
730  pgHandleChild->PGxCONH.bits.SOCS = 0b1111;
731  pgHandleChild->PGxSPCIL.bits.PSS = 0b00001; // Internally connected to the output of PWMPCI[2:0] MUX
732  pgHandleChild->PGxLEBH.bits.PWMPCI = pgInstance;
733  }
734 
735 
736  return(retval);
737 
738 }
